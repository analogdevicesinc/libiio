28 static inline int translate_wsa_error_to_posix(int wsa_err)
29 {
30  switch (wsa_err) {
31  case WSAEACCES:
32  return -EACCES;
33  case WSAEADDRINUSE:
34  return -EADDRINUSE;
35  case WSAEADDRNOTAVAIL:
36  return -EADDRNOTAVAIL;
37  case WSAEAFNOSUPPORT:
38  return -EAFNOSUPPORT;
39  case WSAEALREADY:
40  return -EALREADY;
41  case WSAEBADF:
42  return -EBADF;
43  case WSAECONNREFUSED:
44  return -ECONNREFUSED;
45  case WSAECONNRESET:
46  return -ECONNRESET;
47  case WSAECONNABORTED:
48  return -ECONNABORTED;
49  case WSAEDESTADDRREQ:
50  return -EDESTADDRREQ;
51  case WSAEDISCON:
52  return -ECONNRESET; // Graceful shutdown, closest equivalent
53 #ifdef EDQUOT
54  case WSAEDQUOT:
55  return -EDQUOT;
56 #else
57  case WSAEDQUOT:
58  return -ENOSPC; // Fallback to "no space left on device"
59 #endif
60  case WSAEFAULT:
61  return -EFAULT;
62 #ifdef EHOSTDOWN
63  case WSAEHOSTDOWN:
64  return -EHOSTDOWN;
65 #else
66  case WSAEHOSTDOWN:
67  return -ENETUNREACH; // Fallback to network unreachable
68 #endif
69  case WSAEHOSTUNREACH:
70  return -EHOSTUNREACH;
71  case WSAEINPROGRESS:
72  return -EINPROGRESS;
73  case WSAEINTR:
74  return -EINTR;
75  case WSAEINVAL:
76  return -EINVAL;
77  case WSAEISCONN:
78  return -EISCONN;
79  case WSAELOOP:
80  return -ELOOP;
81  case WSAEMFILE:
82  return -EMFILE;
83  case WSAEMSGSIZE:
84  return -EMSGSIZE;
85  case WSAENAMETOOLONG:
86  return -ENAMETOOLONG;
87  case WSAENETDOWN:
88  return -ENETDOWN;
89  case WSAENETRESET:
90  return -ENETRESET;
91  case WSAENETUNREACH:
92  return -ENETUNREACH;
93  case WSAENOBUFS:
94  return -ENOBUFS;
95  case WSAENOPROTOOPT:
96  return -ENOPROTOOPT;
97  case WSAENOTEMPTY:
98  return -ENOTEMPTY;
99  case WSAENOTSOCK:
100  return -ENOTSOCK;
101  case WSAENOTCONN:
102  return -ENOTCONN;
103  case WSAEOPNOTSUPP:
104  return -EOPNOTSUPP;
105 #ifdef EPFNOSUPPORT
106  case WSAEPFNOSUPPORT:
107  return -EPFNOSUPPORT;
108 #else
109  case WSAEPFNOSUPPORT:
110  return -EAFNOSUPPORT; // Fallback to address family not supported
111 #endif
112  case WSAEPROCLIM:
113  return -EAGAIN; // Too many processes, closest equivalent
114  case WSAEPROTONOSUPPORT:
115  return -EPROTONOSUPPORT;
116  case WSAEPROTOTYPE:
117  return -EPROTOTYPE;
118 #ifdef EREMOTE
119  case WSAEREMOTE:
120  return -EREMOTE;
121 #else
122  case WSAEREMOTE:
123  return -EIO; // Fallback to I/O error
124 #endif
125 #ifdef ESHUTDOWN
126  case WSAESHUTDOWN:
127  return -ESHUTDOWN;
128 #else
129  case WSAESHUTDOWN:
130  return -ECONNABORTED;
131 #endif
132 #ifdef ESOCKTNOSUPPORT
133  case WSAESOCKTNOSUPPORT:
134  return -ESOCKTNOSUPPORT;
135 #else
136  case WSAESOCKTNOSUPPORT:
137  return -EPROTONOSUPPORT; // Fallback to protocol not supported
138 #endif
139 #ifdef ESTALE
140  case WSAESTALE:
141  return -ESTALE;
142 #else
143  case WSAESTALE:
144  return -EIO; // Fallback to I/O error
145 #endif
146  case WSAETIMEDOUT:
147  return -ETIMEDOUT;
148 #ifdef ETOOMANYREFS
149  case WSAETOOMANYREFS:
150  return -ETOOMANYREFS;
151 #else
152  case WSAETOOMANYREFS:
153  return -ENOBUFS; // Fallback to closest equivalent
154 #endif
155 #ifdef EUSERS
156  case WSAEUSERS:
157  return -EUSERS;
158 #else
159  case WSAEUSERS:
160  return -EAGAIN; // Fallback for user quota exceeded
161 #endif
162  case WSAEWOULDBLOCK:
163  return -EAGAIN;
164  default:
165  if (wsa_err > -4096 && wsa_err < 0) // pass through for POSIX errors
166  return wsa_err;
167 
168  return -EIO; // generic fallback
169  }
170 }
